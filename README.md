>提交规范：
>
>'build', // 编译相关的修改，例如发布版本、对项目构建或者依赖的改动
>
>'feat', // 新功能
>
>'fix', // 修补bug
>
>'docs', // 文档修改
>
>'style', // 代码格式修改, 注意不是 css 修改
>
>'refactor', // 重构
>
>'perf', // 优化相关，比如提升性能、体验
>
>'test', // 测试用例修改
>
>'revert', // 代码回滚
>
>'ci', // 持续集成修改
>
>'config', // 配置修改
>
>'chore', // 其他改动0412
